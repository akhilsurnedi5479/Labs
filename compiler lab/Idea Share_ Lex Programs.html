1. Write a Lex Program to count number of vowels and consonents.


Solution:-

             }

  (or) shows a simple lex specification to recognize these verbs.

7. Program to recognize a valid arithmetic expression and identify the identifiers and operators present.



solution:-




prabhat@prabhat-HP-Pro-3005-3085:~/Desktop$ pk 6.l



%{

    #include<stdio.h>

    int flag=1;

    int op=0,dig=0,top=-1;

    char st[10];

%}

digit [0-9]+

oper [+*/-]

%%

['(']                  {  st[++top]='(';  }

[')']                  {  flag=1;

                           if((st[top]!='(') && (top!=-1))

                            {

                               printf("\n Invalid Expression");

                               exit(0);

                            }

                             top--;

                        }

{digit}                {  dig++;   }

{oper}/{digit}     {  op++;    }

{oper}/['(']          {  op++;    }

.                         {printf("\nInvalid Expression"); exit(0);}

%%

main()

{

 printf("\nEnter arithmetic expression");

 yylex();

 if(((op+1==dig) || (op==dig)) && (top==-1))

 {

  printf("\nValid Expression");

   printf("\nInvalid expression");

}

Postfix calculator in LEX


%{ /* C declarations used in actions */
 #define stack_size 100
 static int sp, stack [stack_size];
 static void push (int i) {
   if (++sp<stack_size) stack[sp]= i;
   else {printf ("error: stack overflow\n"); exit(1);}
 }
 static int pop (void) {
   if (sp>=0) return stack[sp--];
   else {printf ("error: stack underflow\n"); exit(1);}
 }
 %}

 /*lex definitions - not used in this example*/

 %%

 /* descriptions of       corresponding actions
    expected inputs       (C statements or blocks) */

 [0-9]+                   {push (atoi(yytext));}
 "+"                      {push (pop() + pop());}
 "*"                      {push (pop() * pop());}
 -                        {int rhs= pop(); push (pop() - rhs);}
 "/"                      {int rhs= pop(); push (pop() / rhs);}
 ;                        {printf ("%d\n", pop());}
 [ \t\n]                  ;
 [^-0-9+*/; \t\n]+        {ECHO; printf (" unexpected\n");}

 %%                        /* C code */

 int main (void) {sp= -1; yylex(); return 0;}

 static int yywrap (void) {return 1;}
